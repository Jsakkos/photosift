**Goal:** Add AI-assisted culling signal — face detection, eye open/closed state, per-eye and whole-image sharpness — that layers on top of the existing three-pass UX without replacing human decisions.

**Tech Stack:** Rust, `ort` 2.x, `ndarray`, `image` 0.25, SQLite via `rusqlite`, React 19, Zustand, TypeScript, Vitest, Tailwind.

```rust
                photo_id, bbox_x, bbox_y, bbox_w, bbox_h,

            stmt.execute(params![

                f.left_eye_x, f.left_eye_y, f.right_eye_x, f.right_eye_y,

/// Write AI aggregates + timestamp in a single call. Pass None for face_count

    rows.collect()
}
```

```rust
/// Discrete Laplacian kernel: [[0,1,0],[1,-4,1],[0,1,0]]
/// Variance of the filtered image is the "sharpness" proxy. Higher = sharper.
pub fn laplacian_variance(img: &GrayImage) -> f64 {
    let (w, h) = img.dimensions();
    if w < 3 || h < 3 { return 0.0; }

    let get = |x: u32, y: u32| img.get_pixel(x, y).0[0] as f64;
    let n = ((w - 2) * (h - 2)) as f64;

    let mut sum = 0.0_f64;
    let mut sum_sq = 0.0_f64;
    for y in 1..h - 1 {
        for x in 1..w - 1 {
            let lap = get(x, y - 1) + get(x - 1, y) + get(x + 1, y) + get(x, y + 1)
                - 4.0 * get(x, y);
            sum += lap;
            sum_sq += lap * lap;
        }
    }
    let mean = sum / n;
    (sum_sq / n) - (mean * mean)
}

/// Compute Laplacian variance for each tile in a `cols × rows` grid.
/// Returns row-major flat vec, length `cols * rows`.
pub fn tiled_laplacian(img: &GrayImage, cols: u32, rows: u32) -> Vec<f64> {
    let (w, h) = img.dimensions();
    let tw = w / cols;
    let th = h / rows;
    let mut out = Vec::with_capacity((cols * rows) as usize);
    for ry in 0..rows {
        for cx in 0..cols {
            let x0 = cx * tw;
            let y0 = ry * th;
            let sub = image::imageops::crop_imm(img, x0, y0, tw, th).to_image();
            out.push(laplacian_variance(&sub));
        }
    }
    out
}

/// Empirical constant tuned against D750 NEF previews — a Laplacian variance
/// of 500 on a 1024px image is "pretty sharp." Adjust during Task 18 if real
/// fixtures indicate a different range.
pub const CALIBRATION_FULL_SCALE: f64 = 500.0;

pub fn normalize_sharpness(variance: f64) -> f64 {
    if variance <= 0.0 { return 0.0; }
    let scaled = variance / CALIBRATION_FULL_SCALE * 100.0;
    if scaled > 100.0 { 100.0 } else { scaled }
}
```

```rust
/// width, clamped to image bounds. All inputs are normalized 0-1; output is
/// pixel coordinates on an `img_w × img_h` image.
pub fn eye_crop_pixels(
    eye: &NormPoint,
    face: &NormBox,
    img_w: u32,
    img_h: u32,
) -> PixelCrop {
    let face_w_px = face.w * img_w as f64;
    let side_px = (face_w_px * 0.15).max(8.0) as i32;
    let half = side_px / 2;

    let cx = (eye.x * img_w as f64) as i32;
    let cy = (eye.y * img_h as f64) as i32;

    let mut x = cx - half;
    let mut y = cy - half;
    let mut w = side_px;
    let mut h = side_px;

    if x < 0 { w += x; x = 0; }
    if y < 0 { h += y; y = 0; }
    if x + w > img_w as i32 { w = img_w as i32 - x; }
    if y + h > img_h as i32 { h = img_h as i32 - y; }
    let side = w.min(h).max(1);

    PixelCrop { x: x as u32, y: y as u32, w: side as u32, h: side as u32 }
}
```

```rust
/// Process a single job end-to-end. Public for test access.
pub fn process_job(
    db: &mut Database,
    job: &AiJob,
    faces_provider: &dyn FaceProvider,
    eyes_provider: &dyn EyeStateProvider,
) -> Result<()> {
    let preview_path = Path::new(&job.preview_path);
    let img = image::open(preview_path)
        .with_context(|| format!("open preview {}", job.preview_path))?;
    let (img_w, img_h) = img.dimensions();
    let gray = img.to_luma8();

    let faces = faces_provider.detect(&gray)?;

    let mut face_rows = Vec::with_capacity(faces.len());
    let mut open_count = 0;
    for face in &faces {
        let l_crop = eye_crop_pixels(&face.left_eye, &face.bbox, img_w, img_h);
        let r_crop = eye_crop_pixels(&face.right_eye, &face.bbox, img_w, img_h);
        let l_img = image::imageops::crop_imm(&gray, l_crop.x, l_crop.y, l_crop.w, l_crop.h).to_image();
        let r_img = image::imageops::crop_imm(&gray, r_crop.x, r_crop.y, r_crop.w, r_crop.h).to_image();

        let l_open = eyes_provider.classify(&l_img)?;
        let r_open = eyes_provider.classify(&r_img)?;
        open_count += l_open + r_open;

        let l_sharp = normalize_sharpness(laplacian_variance(&l_img));
        let r_sharp = normalize_sharpness(laplacian_variance(&r_img));

        face_rows.push(FaceRow {
            photo_id: job.photo_id,
            bbox_x: face.bbox.x, bbox_y: face.bbox.y,
            bbox_w: face.bbox.w, bbox_h: face.bbox.h,
            left_eye_x: face.left_eye.x, left_eye_y: face.left_eye.y,
            right_eye_x: face.right_eye.x, right_eye_y: face.right_eye.y,
            left_eye_open: l_open, right_eye_open: r_open,
            left_eye_sharpness: l_sharp, right_eye_sharpness: r_sharp,
            detection_confidence: face.confidence,
        });
    }

    // Whole-image sharpness on the already-decoded gray image.
    let whole = normalize_sharpness(laplacian_variance(&gray));

    db.delete_faces_for_photo(job.photo_id)?;
    if !face_rows.is_empty() {
        db.insert_faces_batch(&face_rows)?;
    }
    db.mark_ai_analyzed(
        job.photo_id,
        Some(faces.len() as i32),
        Some(open_count),
        whole,
    )?;
    Ok(())
}

/// Main worker loop. Pulls jobs until the channel closes or the cancel flag
/// flips. On cancel, drops remaining queued jobs.
pub fn run_loop(
    rx: Receiver<AiJob>,
    cancel: Arc<AtomicBool>,
    mut db: Database,
    faces_provider: Box<dyn FaceProvider>,
    eyes_provider: Box<dyn EyeStateProvider>,
    on_progress: impl Fn(&AiJob, Result<()>) + Send,
) {
    while let Ok(job) = rx.recv() {
        if cancel.load(Ordering::SeqCst) {
            break;
        }
        let result = process_job(&mut db, &job, faces_provider.as_ref(), eyes_provider.as_ref());
        on_progress(&job, result);
    }
}
```

```rust
use crate::ai::face::FaceProvider;
use crate::ai::eye::EyeStateProvider;
use crate::db::schema::Database;
use crossbeam_channel::unbounded;
use std::sync::Arc;
use std::sync::atomic::{AtomicBool, AtomicUsize};
use std::thread;

pub struct SpawnedWorker {
    pub handle: worker::WorkerHandle,
}

pub fn spawn_worker(
    db_path: std::path::PathBuf,
    faces_provider: Box<dyn FaceProvider>,
    eyes_provider: Box<dyn EyeStateProvider>,
    cancel: Arc<AtomicBool>,
    analyzed: Arc<AtomicUsize>,
    failed: Arc<AtomicUsize>,
    on_job_done: impl Fn(i64, bool) + Send + 'static,
) -> SpawnedWorker {
    let (tx, rx) = unbounded::<AiJob>();
    let cancel_clone = cancel.clone();
    thread::spawn(move || {
        let db = match Database::open(&db_path) {
            Ok(d) => d,
            Err(e) => { log::error!("ai worker db open failed: {}", e); return; }
        };
        worker::run_loop(rx, cancel_clone, db, faces_provider, eyes_provider, move |job, res| {
            match res {
                Ok(_) => { analyzed.fetch_add(1, std::sync::atomic::Ordering::SeqCst); on_job_done(job.photo_id, true); }
                Err(e) => {
                    log::error!("ai job failed for photo {}: {}", job.photo_id, e);
                    failed.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
                    on_job_done(job.photo_id, false);
                }
            }
        });
    });
    SpawnedWorker {
        handle: worker::WorkerHandle { sender: tx, cancel },
    }
}
```

```ts
export function aiPickForGroup(group: Group, images: ImageEntry[]): number | null {
  const analyzed = group.members
    .map((m) => images.find((i) => i.id === m.photoId))
    .filter((img): img is ImageEntry => !!img && img.aiAnalyzedAt != null);

  if (analyzed.length < 2) return null;

  let best = analyzed[0];
  let bestScore = (best.sharpnessScore ?? 0) * (1 + (best.eyesOpenCount ?? 0));
  for (const img of analyzed.slice(1)) {
    const score = (img.sharpnessScore ?? 0) * (1 + (img.eyesOpenCount ?? 0));
    if (score > bestScore || (score === bestScore && img.id < best.id)) {
      best = img;
      bestScore = score;
    }
  }
  return best.id;
}

export function acceptAiPick(): Promise<void> {
  const { groups, images, displayItems, currentIndex, setGroupCover } = useProjectStore.getState();
  const item = displayItems[currentIndex];
  if (!item?.groupId) return Promise.resolve();
  const group = groups.find((g) => g.id === item.groupId);
  if (!group) return Promise.resolve();
  const pickId = aiPickForGroup(group, images);
  if (pickId === null) return Promise.resolve();
  return setGroupCover(group.id, pickId);
}
```

```tsx
  const heatmapOn = useProjectStore((s) => s.heatmapOn);
  const getData = useProjectStore((s) => s.getHeatmapData);
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    if (!heatmapOn) return;
    const data = getData(photoId);
    if (!data) {
      // Re-render when cache populates

        if (state.heatmapCache.has(photoId)) {
          draw(canvasRef.current, state.heatmapCache.get(photoId)!);
          unsub();
        }
      });
      return unsub;
    }
    draw(canvasRef.current, data);
  }, [heatmapOn, photoId, getData]);

  return (

      ref={canvasRef}
      width={32}
      height={32}
      aria-hidden="true"

      style={{ imageRendering: "pixelated" }}
    />

function draw(canvas: HTMLCanvasElement | null, data: number[]) {
  if (!canvas) return;
  const ctx = canvas.getContext("2d");
  if (!ctx) return;
  const img = ctx.createImageData(32, 32);
  for (let i = 0; i < 1024; i++) {
    const v = Math.max(0, Math.min(100, data[i] ?? 0)) / 100;
    // Red→Yellow→Green gradient
    const r = Math.round(255 * (1 - v));
    const g = Math.round(255 * v);
    img.data[i * 4] = r;
    img.data[i * 4 + 1] = g;
    img.data[i * 4 + 2] = 0;
```

Write `YuNetProvider` in `face.rs` using `ort`:

```rust
use ort::session::{builder::GraphOptimizationLevel, Session};
use ort::execution_providers::{CUDAExecutionProvider, CPUExecutionProvider};
use std::path::Path;

pub struct YuNetProvider {
    session: Session,
}

impl YuNetProvider {
    pub fn load(model_path: &Path, try_cuda: bool) -> anyhow::Result<(Self, crate::ai::AiProviderStatus)> {
        let mut builder = Session::builder()?
            .with_optimization_level(GraphOptimizationLevel::Level3)?;
        let status = if try_cuda {
            match builder.with_execution_providers([CUDAExecutionProvider::default().build()]) {
                Ok(b) => { builder = b; crate::ai::AiProviderStatus::Cuda }
                Err(_) => {
                    builder = builder.with_execution_providers([CPUExecutionProvider::default().build()])?;
                    crate::ai::AiProviderStatus::Cpu
                }
            }
        } else {
            builder = builder.with_execution_providers([CPUExecutionProvider::default().build()])?;
            crate::ai::AiProviderStatus::Cpu
        };
        let session = builder.commit_from_file(model_path)?;
        Ok((YuNetProvider { session }, status))
    }
}

impl FaceProvider for YuNetProvider {
    fn detect(&self, gray: &image::GrayImage) -> anyhow::Result<Vec<DetectedFace>> {
        // 1. Letterbox-resize the input gray image to 640×640 (fixed by YuNet
        //    2023mar), tracking the scale + pad so we can map detections back.
        // 2. Build a [1,3,640,640] f32 NCHW tensor (YuNet expects 3-channel
        //    RGB, so duplicate the gray plane into R/G/B).
        // 3. Run the session via `ort::inputs![tensor.view()]` → `session.run`.
        // 4. Parse outputs: YuNet returns three tensors per head × three
        //    heads (cls, obj, bbox+kps). Reference:
        //    https://github.com/opencv/opencv_zoo/tree/main/models/face_detection_yunet
        //    OpenCV's `FaceDetectorYN` C++ source (modules/objdetect/src/face_detect.cpp)
        //    is the authoritative parser; port its anchor/decoding logic.
        // 5. Apply confidence threshold 0.6 and NMS (IoU 0.3).
        // 6. Map kept detections back to original pixel coords via the
        //    letterbox inverse, then normalize to 0–1.
        // 7. Emit DetectedFace { bbox, left_eye (kp 0), right_eye (kp 1), confidence }.
        //
        // This body is ~80–120 lines of tensor math. The TEST in the next
        // step is the acceptance gate — if it passes on the fixture image,
        // the implementation is correct.
        anyhow::bail!("YuNet inference not yet implemented — see step 3b")
    }
}
```
